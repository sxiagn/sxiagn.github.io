# 表单校验不通过，滚动到第一个位置

  • 作为产品，希望表单校验不通过时，页面滚动到第一个不通过校验的表单元素（比如input标签）。
  • 正常情况前端无需额外处理，表单校验不通过会在表单元素底部有文字提示。
  • 多表单元素时，比如整个表单跨度占满一屏幕，校验不通过页面没反应，体验不好。
// 第一步：给需要检验的表单添加都ref属性，名字跟prop绑定的一致
<el-form-item label="活动形式" prop="desc" ref="desc">
	<el-input type="textarea"  v-model="ruleForm.desc"></el-input>
</el-form-item>

// 第二步：validate方法传入第二个参数object
this.$refs[formName].validate((valid, object) => {
    if (valid) {
        console.log('校验通过')
    } else {
        // 传入validate的第二个参数：object
        for (const i in object) {
            let dom = this.$refs[i]
            if (Object.prototype.toString.call(dom) !== '[object Object]') { 
                // 这里是针对遍历的情况（多个输入框），取值为数组
                dom = dom[0]
            }
            // 滚动到指定节点
            dom.$el.scrollIntoView({
                block: 'center', // 值有start,center,end，nearest，当前显示在视图区域中间
                behavior: 'smooth' // 值有auto、instant,smooth，缓动动画（当前是慢速的）
            })
            break // 因为我们只需要检测一项,所以就可以跳出循环了
        }
    }
})
