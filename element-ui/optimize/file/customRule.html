    # 自定义表单校验规则

    // 示列表单html结构
<el-form :model="formData" :rules="rulesForm" ref="ruleForm" class="demo-ruleForm">
    <el-form-item label="姓名" prop="name">
        <el-input v-model="formData.name"></el-input>
    </el-form-item>
    <el-form-item>
        <el-button type="primary" @click="submitForm('ruleForm')">确认</el-button>
	</el-form-item>
</el-form>

// 表单校验规则建议放在computed（放在data，在国际化项目切换语言时data的值不更新）
computed: {
    rulesForm() {
      var nameRule = (rule, value, callback) => {
        if (!value) {
          callback(new Error('请输入姓名'))
        } else {
          callback()
        }
      }
      return {
        name: [
          { required: true, validator: nameRule, trigger: 'blur' }
          // { min: 3, max: 5, message: '长度在 3 到 5 个字符', trigger: ['change', 'blur'] }
        ]
      }
    }
  }
    # 参数说明
    • required为布尔值，控制表单元素是否必填
    • validater为自定义方法
    • trigger为校验的触发方式，blur为失焦，可以有多种触发方式
    • trigger: ['blur', 'change'] 指的是失焦或者change事件都会触发校验
    • 当然，nameRule也可以当成方法写在methods里面，这样调用 validator: this.nameRule，不需要加()
