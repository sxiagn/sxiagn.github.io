# 自定义icon图标

// 准备自定义icon的样式
.el-icon-new{
    background: url('./assets/images/small.png') center no-repeat; 
    // 图标路径，最好准备png格式
    background-size: cover;
 }
 .el-icon-new:before{
    content: '';
    font-size: 24px; // 控制图标的大小
    visibility: hidden; // 元素占位置，但不可见
 }
 
// 在html中使用
<el-button icon="el-icon-new"></el-button>
// 或者
<i icon="el-icon-new"></i>
